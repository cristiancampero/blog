<h2>Creation of the Database</h2>

```sql
--Create the database'
CREATE DATABASE 'platziblog' DEFAULT CHARACTER SET utf8 ;
```
<br />


```sql
--Select the database'
USE platziblog; 
```
<br />


<h2>Creation of tables</h2>

```sql
--Creating the table 'usuarios'
CREATE TABLE 'platziblog'.'usuarios' (
  'id' INT NOT NULL AUTO_INCREMENT,
  'login' VARCHAR(30) NOT NULL,
  'password' VARCHAR(32) NOT NULL,
  'nickname' VARCHAR(40) NOT NULL,
  'email' VARCHAR(40) NOT NULL,
  PRIMARY KEY ('id'),
  UNIQUE INDEX 'email_UNIQUE' ('email' ASC) VISIBLE,
  UNIQUE INDEX 'nickname_UNIQUE' ('nickname' ASC) VISIBLE);
```
```sql
--Creating the table 'categorias'
CREATE TABLE 'platziblog'.'categorias' (
  'id' INT NOT NULL AUTO_INCREMENT,
  'nombre_categoria' VARCHAR(30) NOT NULL,
  PRIMARY KEY ('id'));
```
```sql
--Creating the table 'etiquetas'
CREATE TABLE 'platziblog'.'etiquetas' (  
  'id' INT NOT NULL AUTO_INCREMENT,      
  'nombre_etiqueta' VARCHAR(30) NOT NULL,
  PRIMARY KEY ('id'));
```

```sql
--Creating the table 'posts'
CREATE TABLE 'platziblog'.'posts' (
  'id' INT NOT NULL AUTO_INCREMENT,
  'titulo' VARCHAR(130) NOT NULL,
  'fecha_publicacion' TIMESTAMP NULL,
  'contenido' TEXT NOT NULL,
  'estatus' CHAR(8) NULL DEFAULT 'activo',
  'usuario_id' INT NOT NULL,
  'categoria_id' INT NOT NULL,
  PRIMARY KEY ('id'),
  INDEX 'posts_usuarios_idx' ('usuario_id' ASC) VISIBLE,    
  INDEX 'posts_categorias_idx' ('categoria_id' ASC) VISIBLE,
  CONSTRAINT 'posts_usuarios'
    FOREIGN KEY ('usuario_id')
    REFERENCES 'platziblog'.'usuarios' ('id')
    ON DELETE NO ACTION
    ON UPDATE CASCADE,
  CONSTRAINT 'posts_categorias'
    FOREIGN KEY ('categoria_id')
    REFERENCES 'platziblog'.'categorias' ('id')
    ON DELETE NO ACTION
    ON UPDATE NO ACTION);
```
```sql
--Creating the table 'comentarios'
CREATE TABLE 'platziblog'.'comentarios' (
  'id' INT NOT NULL AUTO_INCREMENT,
  'cuerpo_comentario' TEXT NOT NULL,
  'usuario_id' INT NOT NULL,
  'post_id' INT NOT NULL,
  PRIMARY KEY ('id'),
  INDEX 'comentarios_usuarios_idx' ('usuario_id' ASC) VISIBLE,
  INDEX 'comentarios_post_idx' ('post_id' ASC) VISIBLE,       
  CONSTRAINT 'comentarios_usuario'
    FOREIGN KEY ('usuario_id')
    REFERENCES 'platziblog'.'usuarios' ('id')
    ON DELETE NO ACTION
    ON UPDATE NO ACTION,
  CONSTRAINT 'comentarios_post'
    FOREIGN KEY ('post_id')
    REFERENCES 'platziblog'.'posts' ('id')
    ON DELETE NO ACTION
    ON UPDATE NO ACTION);
```
```sql
--Creating the table 'posts_etiquetas'
CREATE TABLE 'platziblog'.'posts_etiquetas' (
  'id' INT NOT NULL AUTO_INCREMENT,
  'post_id' INT NOT NULL,
  'etiqueta_id' INT NOT NULL,
  PRIMARY KEY ('id'),
  INDEX 'postetiquetas_post_idx' ('post_id' ASC) VISIBLE,
  INDEX 'postetiquetas_etiquetas_idx' ('etiqueta_id' ASC) VISIBLE,
  CONSTRAINT 'postetiquetas_post'
    FOREIGN KEY ('post_id')
    REFERENCES 'platziblog'.'posts' ('id')
    ON DELETE NO ACTION
    ON UPDATE NO ACTION,
  CONSTRAINT 'postetiquetas_etiquetas'
    FOREIGN KEY ('etiqueta_id')
    REFERENCES 'platziblog'.'etiquetas' ('id')
    ON DELETE NO ACTION
    ON UPDATE NO ACTION);
```

--Show title and number of tags

--from table 'posts'
FROM posts
--join table 'posts' with table 'posts_etiquetas' through their keys
INNER JOIN posts_etiquetas ON posts.id = posts_etiquetas.post_id
--join table 'posts_etiquetas' with table 'etiquetas' through their keys
INNER JOIN etiquetas ON etiquetas.id = posts_etiquetas.etiqueta_id
--group by post id
GROUP BY posts.id;
--sort by tag number from highest to lowest
ORDER BY num_etiquetas DESC;
